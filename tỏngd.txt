local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local Camera = workspace.CurrentCamera
local player = Players.LocalPlayer

-- [[ BẢNG TRẠNG THÁI HỆ THỐNG ]] --
_G.PVP_Settings = {
    EspLine = false,
    HitboxEnabled = false,
    HitboxSize = 25
}

-- [[ 1. TẠO GIAO DIỆN (UI) ]] --
local gui = Instance.new("ScreenGui", player.PlayerGui)
gui.Name = "Migul_V4_Final"
gui.ResetOnSpawn = false

local mainFrame = Instance.new("Frame", gui)
mainFrame.Size = UDim2.new(0, 350, 0, 400)
mainFrame.Position = UDim2.new(0.5, -175, 0.5, -200)
mainFrame.BackgroundColor3 = Color3.fromRGB(15, 17, 24) -- Dark Theme
mainFrame.Visible = false -- Mặc định ẩn, bấm nút MIGUL mới hiện
Instance.new("UICorner", mainFrame).CornerRadius = UDim.new(0, 12)

local scroll = Instance.new("ScrollingFrame", mainFrame)
scroll.Size = UDim2.new(1, -20, 1, -60)
scroll.Position = UDim2.new(0, 10, 0, 50)
scroll.BackgroundTransparency = 1
scroll.ScrollBarThickness = 2
Instance.new("UIListLayout", scroll).Padding = UDim.new(0, 8)

-- [[ 2. NÚT ĐÓNG/MỞ (FLOATING BUTTON) ]] --
local openBtn = Instance.new("TextButton", gui)
openBtn.Size = UDim2.new(0, 50, 0, 50)
openBtn.Position = UDim2.new(0, 15, 0, 15)
openBtn.BackgroundColor3 = Color3.fromRGB(0, 180, 255)
openBtn.Text = "MIGUL"
openBtn.TextColor3 = Color3.new(1, 1, 1)
openBtn.Font = Enum.Font.GothamBold
openBtn.TextSize = 10
Instance.new("UICorner", openBtn).CornerRadius = UDim.new(1, 0)
openBtn.Draggable = true -- Có thể kéo nút

openBtn.MouseButton1Click:Connect(function()
    mainFrame.Visible = not mainFrame.Visible
    openBtn.BackgroundColor3 = mainFrame.Visible and Color3.fromRGB(255, 50, 50) or Color3.fromRGB(0, 180, 255)
end)

-- [[ 3. HÀM TẠO TOGGLE (BẬT/TẮT) ]] --
local function CreateToggle(name, callback)
    local btn = Instance.new("TextButton", scroll)
    btn.Size = UDim2.new(1, -5, 0, 40)
    btn.BackgroundColor3 = Color3.fromRGB(30, 33, 45)
    btn.Text = "[ OFF ] " .. name
    btn.TextColor3 = Color3.new(1, 1, 1)
    btn.Font = Enum.Font.Gotham
    Instance.new("UICorner", btn).CornerRadius = UDim.new(0, 8)

    local active = false
    btn.MouseButton1Click:Connect(function()
        active = not active
        btn.BackgroundColor3 = active and Color3.fromRGB(0, 180, 255) or Color3.fromRGB(30, 33, 45)
        btn.Text = active and "[ ON ] " .. name or "[ OFF ] " .. name
        callback(active)
    end)
end

-- [[ 4. LOGIC ESP LINE (ĐỈNH MÀN HÌNH) ]] --
local function CreateESPLine(target)
    local Line = Drawing.new("Line")
    RunService.RenderStepped:Connect(function()
        if target.Character and target.Character:FindFirstChild("HumanoidRootPart") and target.Character.Humanoid.Health > 0 then
            local Pos, OnScreen = Camera:WorldToViewportPoint(target.Character.HumanoidRootPart.Position)
            if OnScreen and _G.PVP_Settings.EspLine then
                Line.Visible = true
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0) -- Từ đỉnh màn hình
                Line.To = Vector2.new(Pos.X, Pos.Y)
                Line.Color = Color3.fromRGB(0, 255, 255)
                Line.Thickness = 1
            else Line.Visible = false end
        else Line.Visible = false end
    end)
end

for _, v in pairs(Players:GetPlayers()) do if v ~= player then CreateESPLine(v) end end
Players.PlayerAdded:Connect(function(v) if v ~= player then CreateESPLine(v) end end)

-- [[ 5. LOGIC HITBOX ]] --
RunService.RenderStepped:Connect(function()
    if _G.PVP_Settings.HitboxEnabled then
        for _, target in pairs(Players:GetPlayers()) do
            if target ~= player and target.Character and target.Character:FindFirstChild("HumanoidRootPart") then
                pcall(function()
                    local hrp = target.Character.HumanoidRootPart
                    hrp.Size = Vector3.new(_G.PVP_Settings.HitboxSize, _G.PVP_Settings.HitboxSize, _G.PVP_Settings.HitboxSize)
                    hrp.Transparency = 0.7
                    hrp.Color = Color3.fromRGB(0, 180, 255)
                    hrp.Material = Enum.Material.Neon
                    hrp.CanCollide = false
                end)
            end
        end
    end
end)

end)

end)
