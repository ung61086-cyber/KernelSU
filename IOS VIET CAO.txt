-- [[ SERVICES ]] --
local Players = game:GetService("Players")
local RunService = game:GetService("RunService")
local LocalPlayer = Players.LocalPlayer
local Camera = workspace.CurrentCamera

-- [[ CẤU HÌNH HỆ THỐNG ]] --
_G.Config = {
    ESP_Master = true,
    ESP_Box = true,
    ESP_Line = true,
    ESP_HP = true,
    TeamCheck = true,
    AimRadius = 93.61, -- Thông số từ video
}

local Cache = { Lines = {}, Boxes = {}, HealthBars = {} }
local Keys = {
    ["tudeptrai2013"] = 863913600, -- VIP 9999 ngày
    ["Tudeptrai"] = 86400,
    ["Tudeptrai09"] = 86400,
    ["Tudeptrai10"] = 86400, -- Thêm key mới
    ["Tudeptrai33"] = 86400,
    ["Tudeptrai88"] = 86400,
    ["Tudeptrai98"] = 86400
}

-- [[ HÀM TẠO VẼ (DRAWING) ]] --
local function CreateESP(player)
    if Cache.Lines[player.Name] then return end -- Tránh tạo trùng

    local line = Drawing.new("Line")
    line.Thickness = 1
    line.Color = Color3.new(1, 1, 1)
    line.Transparency = 1
    Cache.Lines[player.Name] = line

    local box = {
        T = Drawing.new("Line"), B = Drawing.new("Line"), 
        L = Drawing.new("Line"), R = Drawing.new("Line")
    }
    for _, l in pairs(box) do 
        l.Thickness = 1.5
        l.Color = Color3.new(1, 1, 1)
        l.Visible = false
    end
    Cache.Boxes[player.Name] = box

    local hb = { Out = Drawing.new("Square"), Main = Drawing.new("Square") }
    hb.Out.Filled = true
    hb.Out.Color = Color3.fromRGB(0,0,0)
    hb.Out.Transparency = 0.5
    hb.Main.Filled = true
    hb.Main.Transparency = 1
    Cache.HealthBars[player.Name] = hb
end

-- [[ HÀM DỌN DẸP KHI PLAYER THOÁT ]] --
local function RemoveESP(playerName)
    if Cache.Lines[playerName] then Cache.Lines[playerName]:Remove() end
    if Cache.Boxes[playerName] then 
        for _, l in pairs(Cache.Boxes[playerName]) do l:Remove() end 
    end
    if Cache.HealthBars[playerName] then
        Cache.HealthBars[playerName].Out:Remove()
        Cache.HealthBars[playerName].Main:Remove()
    end
    Cache.Lines[playerName] = nil
    Cache.Boxes[playerName] = nil
    Cache.HealthBars[playerName] = nil
end

-- [[ LOGIC HIỂN THỊ ESP ]] --
local function UpdateESP()
    for _, v in pairs(Players:GetPlayers()) do
        if v == LocalPlayer then continue end
        
        local char = v.Character
        local hrp = char and char:FindFirstChild("HumanoidRootPart")
        local hum = char and char:FindFirstChild("Humanoid")
        
        -- Tạo ESP nếu chưa có
        if hrp and not Cache.Lines[v.Name] then CreateESP(v) end
        
        local canShow = _G.Config.ESP_Master and hrp and hum and hum.Health > 0
        if _G.Config.TeamCheck and v.Team == LocalPlayer.Team then canShow = false end

        local pos, onScreen = false, false
        if canShow then
            pos, onScreen = Camera:WorldToViewportPoint(hrp.Position)
        end

        if onScreen and canShow then
            local sizeX = 2000 / pos.Z
            local sizeY = 3000 / pos.Z
            local x, y = pos.X - sizeX/2, pos.Y - sizeY/2

            -- Cập nhật Box
            local b = Cache.Boxes[v.Name]
            if b then
                b.T.Visible = _G.Config.ESP_Box; b.T.From = Vector2.new(x, y); b.T.To = Vector2.new(x + sizeX, y)
                b.B.Visible = _G.Config.ESP_Box; b.B.From = Vector2.new(x, y + sizeY); b.B.To = Vector2.new(x + sizeX, y + sizeY)
                b.L.Visible = _G.Config.ESP_Box; b.L.From = Vector2.new(x, y); l.L.To = Vector2.new(x, y + sizeY)
                b.R.Visible = _G.Config.ESP_Box; b.R.From = Vector2.new(x + sizeX, y); b.R.To = Vector2.new(x + sizeX, y + sizeY)
            end

            -- Cập nhật Line
            if Cache.Lines[v.Name] then
                Cache.Lines[v.Name].Visible = _G.Config.ESP_Line
                Cache.Lines[v.Name].From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                Cache.Lines[v.Name].To = Vector2.new(pos.X, pos.Y)
            end

            -- Cập nhật Thanh Máu (HP Bar)
            local h = Cache.HealthBars[v.Name]
            if h then
                local hpPct = math.clamp(hum.Health / hum.MaxHealth, 0, 1)
                h.Out.Visible = _G.Config.ESP_HP
                h.Out.Size = Vector2.new(4, sizeY)
                h.Out.Position = Vector2.new(x - 7, y)
                
                h.Main.Visible = _G.Config.ESP_HP
                h.Main.Size = Vector2.new(2, sizeY * hpPct)
                h.Main.Position = Vector2.new(x - 6, y + (sizeY * (1 - hpPct)))
                h.Main.Color = Color3.fromHSV(hpPct * 0.3, 1, 1) -- Màu từ đỏ sang xanh
            end
        else
            -- Ẩn đi khi không nhìn thấy
            if Cache.Lines[v.Name] then Cache.Lines[v.Name].Visible = false end
            if Cache.Boxes[v.Name] then
                for _, l in pairs(Cache.Boxes[v.Name]) do l.Visible = false end
            end
            if Cache.HealthBars[v.Name] then
                Cache.HealthBars[v.Name].Out.Visible = false
                Cache.HealthBars[v.Name].Main.Visible = false
            end
        end
    end
end

-- Dọn dẹp bộ nhớ khi có người thoát
Players.PlayerRemoving:Connect(function(player)
    RemoveESP(player.Name)
end)
