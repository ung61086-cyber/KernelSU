_G.Settings = {

local Camera = workspace.CurrentCamera
local Players = game:GetService("Players")

local RunService = game:GetService("RunService")

-- HÀM TẠO ĐỒ HỌA (DRAWING API)
local function CreateLine(Color)
    local L = Drawing.new("Line")
    L.Thickness = 1
    L.Color = Color or Color3.fromRGB(0, 255, 0)
    L.Transparency = 1
    return L
end

-- LOGIC ESP TỔNG HỢP (FIX BOX & SKELETON)
local function ApplyESP(Player)
    local Line = CreateLine()
    local Box = Drawing.new("Square")
    local Name = Drawing.new("Text")
    
    local Skeleton = {
        H_T = CreateLine(Color3.fromRGB(255,255,255)), T_LA = CreateLine(Color3.fromRGB(255,255,255)), 
        LA_LH = CreateLine(Color3.fromRGB(255,255,255)), T_RA = CreateLine(Color3.fromRGB(255,255,255)), 
        RA_RH = CreateLine(Color3.fromRGB(255,255,255)), T_LL = CreateLine(Color3.fromRGB(255,255,255)), 
        LL_LF = CreateLine(Color3.fromRGB(255,255,255)), T_RL = CreateLine(Color3.fromRGB(255,255,255)), 
        RL_RF = CreateLine(Color3.fromRGB(255,255,255))
    }

    RunService.RenderStepped:Connect(function()
        local Char = Player.Character
        if _G.Settings.EnableESP and Char and Char:FindFirstChild("HumanoidRootPart") and Player ~= LocalPlayer and Char.Humanoid.Health > 0 then
            local Root = Char.HumanoidRootPart
            local Pos, OnScreen = Camera:WorldToViewportPoint(Root.Position)

            if OnScreen then
                -- 1. BOX ESP (FIX SIZE)
                local Height = (Camera.ViewportSize.Y / Pos.Z) * 2.5
                local Width = Height * 0.6
                Box.Visible = _G.Settings.BoxESP
                Box.Size = Vector2.new(Width, Height)
                Box.Position = Vector2.new(Pos.X - Width/2, (Pos.Y + _G.Settings.FixEsp) - Height/2)
                Box.Color = Color3.fromRGB(255, 0, 0)

                -- 2. LINE ESP (FIX: START FROM TOP)
                Line.Visible = _G.Settings.LineESP
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, 0) 
                Line.To = Vector2.new(Pos.X, Pos.Y + _G.Settings.FixEsp)

                -- 3. SKELETON ESP (FIXED VISIBILITY)
                if _G.Settings.SkeletonESP then
                    local function Connect(L, P1, P2)
                        local Part1, Part2 = Char:FindFirstChild(P1), Char:FindFirstChild(P2)
                        if Part1 and Part2 then
                            local V1 = Camera:WorldToViewportPoint(Part1.Position)
                            local V2 = Camera:WorldToViewportPoint(Part2.Position)
                            L.Visible = true
                            L.From = Vector2.new(V1.X, V1.Y + _G.Settings.FixEsp)
                            L.To = Vector2.new(V2.X, V2.Y + _G.Settings.FixEsp)
                        else L.Visible = false end
                    end
                    Connect(Skeleton.H_T, "Head", "UpperTorso")
                    -- (Các khớp nối khác tương tự...)
                else
                    for _, l in pairs(Skeleton) do l.Visible = false end
                end
            end
        else
            Box.Visible = false; Line.Visible = false; for _, l in pairs(Skeleton) do l.Visible = false end
        end
    end)
end
