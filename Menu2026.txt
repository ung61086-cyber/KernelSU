-- Biến toàn cục để lưu trạng thái chức năng
_G.Settings = {
    -- Player ESP
    EnableESP = false, LineESP = false, BoxESP = false, SkeletonESP = false, TargetESP = false, DistanceESP = false, FixEsp = 0,
    -- Visual
    NameESP = false, EnemyCounter = false, HealthESP = false, ShowFOV = false, EspDir = false, FovDir = 150,
    -- Aimbot
    Aimbot = false, AimMagnet = false, FOVRadius = 150, HeadshotRate = 100, AimSmooth = 1, AimMode = "Silent Aim", TargetRegion = "Head", TriggerMode = "Shooting"
}

local Camera = workspace.CurrentCamera
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")

-- VÒNG TRÒN FOV (Cố định ở giữa màn hình theo yêu cầu của bạn)
local FOVCircle = Drawing.new("Circle")
FOVCircle.Thickness = 1
FOVCircle.NumSides = 100
FOVCircle.Color = Color3.fromRGB(255, 255, 255)
FOVCircle.Filled = false

RunService.RenderStepped:Connect(function()
    FOVCircle.Visible = _G.Settings.ShowFOV
    FOVCircle.Radius = _G.Settings.FOVRadius
    FOVCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2) -- Cố định ở tâm
end)

-- LOGIC ESP CHI TIẾT
local function ApplyESP(Player)
    local Line = Drawing.new("Line")
    local Box = Drawing.new("Square")
    local Name = Drawing.new("Text")

    RunService.RenderStepped:Connect(function()
        if _G.Settings.EnableESP and Player.Character and Player.Character:FindFirstChild("HumanoidRootPart") and Player ~= LocalPlayer then
            local Root = Player.Character.HumanoidRootPart
            local Pos, OnScreen = Camera:WorldToViewportPoint(Root.Position)

            if OnScreen then
                local Dist = math.floor((Root.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude)
                
                -- Line ESP (Từ chân màn hình đến mục tiêu)
                Line.Visible = _G.Settings.LineESP
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                Line.To = Vector2.new(Pos.X, Pos.Y + _G.Settings.FixEsp)
                Line.Color = Color3.fromRGB(0, 255, 0)

                -- Box ESP
                Box.Visible = _G.Settings.BoxESP
                Box.Size = Vector2.new(2000/Pos.Z, 2500/Pos.Z)
                Box.Position = Vector2.new(Pos.X - Box.Size.X/2, (Pos.Y + _G.Settings.FixEsp) - Box.Size.Y/2)
                Box.Color = Color3.fromRGB(255, 0, 0)

                -- Name & Distance
                Name.Visible = _G.Settings.NameESP
                Name.Text = Player.Name .. " [" .. Dist .. "m]"
                Name.Position = Vector2.new(Pos.X, (Pos.Y + _G.Settings.FixEsp) - (Box.Size.Y/2) - 20)
                Name.Color = Color3.fromRGB(255, 255, 255)
                Name.Center = true
            else
                Line.Visible = false; Box.Visible = false; Name.Visible = false
            end
        else
            Line.Visible = false; Box.Visible = false; Name.Visible = false
        end
    end)
end

-- Tự động chạy cho người chơi mới
for _, v in pairs(Players:GetPlayers()) do ApplyESP(v) end
Players.PlayerAdded:Connect(ApplyESP)
