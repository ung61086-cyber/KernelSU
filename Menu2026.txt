-- CẤU HÌNH BIẾN TOÀN CỤC
_G.Settings = {
    EnableESP = false, LineESP = false, BoxESP = false, SkeletonESP = false,
    DistanceESP = false, NameESP = false, HealthESP = false, FixEsp = 0,
    ShowFOV = false, FOVRadius = 150, Aimbot = false, AimMode = "Silent Aim",
    TargetRegion = "Head"
}

local Camera = workspace.CurrentCamera
local Players = game:GetService("Players")
local LocalPlayer = Players.LocalPlayer
local RunService = game:GetService("RunService")

-- HÀM TẠO ĐỒ HỌA
local function CreateLine()
    local L = Drawing.new("Line")
    L.Thickness = 1
    L.Color = Color3.fromRGB(0, 255, 0)
    L.Transparency = 1
    return L
end

-- VÒNG TRÒN FOV (CỐ ĐỊNH Ở GIỮA)
local FOVCircle = Drawing.new("Circle")
FOVCircle.Thickness = 1
FOVCircle.NumSides = 100
FOVCircle.Color = Color3.fromRGB(255, 255, 255)
FOVCircle.Filled = false

RunService.RenderStepped:Connect(function()
    FOVCircle.Visible = _G.Settings.ShowFOV
    FOVCircle.Radius = _G.Settings.FOVRadius
    FOVCircle.Position = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y / 2)
end)

-- LOGIC ESP TỔNG HỢP
local function ApplyESP(Player)
    local Line = CreateLine()
    local Box = Drawing.new("Square")
    local Name = Drawing.new("Text")
    
    -- Các đoạn xương (Skeleton)
    local Skeleton = {
        H_T = CreateLine(), T_LA = CreateLine(), LA_LH = CreateLine(),
        T_RA = CreateLine(), RA_RH = CreateLine(), T_LL = CreateLine(),
        LL_LF = CreateLine(), T_RL = CreateLine(), RL_RF = CreateLine()
    }

    RunService.RenderStepped:Connect(function()
        local Char = Player.Character
        if _G.Settings.EnableESP and Char and Char:FindFirstChild("HumanoidRootPart") and Player ~= LocalPlayer and Char.Humanoid.Health > 0 then
            local Root = Char.HumanoidRootPart
            local Pos, OnScreen = Camera:WorldToViewportPoint(Root.Position)

            if OnScreen then
                -- 1. BOX ESP (FIX SIZE - KHÔNG CHE TẦM NHÌN)
                local Height = (Camera.ViewportSize.Y / Pos.Z) * 2.5
                local Width = Height * 0.6
                Box.Visible = _G.Settings.BoxESP
                Box.Size = Vector2.new(Width, Height)
                Box.Position = Vector2.new(Pos.X - Width/2, (Pos.Y + _G.Settings.FixEsp) - Height/2)
                Box.Color = Color3.fromRGB(255, 0, 0)
                Box.Thickness = 1

                -- 2. LINE ESP
                Line.Visible = _G.Settings.LineESP
                Line.From = Vector2.new(Camera.ViewportSize.X / 2, Camera.ViewportSize.Y)
                Line.To = Vector2.new(Pos.X, Pos.Y + _G.Settings.FixEsp)

                -- 3. NAME & DISTANCE
                Name.Visible = _G.Settings.NameESP
                local Dist = math.floor((Root.Position - LocalPlayer.Character.HumanoidRootPart.Position).Magnitude)
                Name.Text = Player.Name .. " [" .. Dist .. "m]"
                Name.Position = Vector2.new(Pos.X, (Pos.Y + _G.Settings.FixEsp) - (Height/2) - 20)
                Name.Center = true; Name.Size = 14; Name.Outline = true

                -- 4. SKELETON ESP (KHUNG XƯƠNG)
                if _G.Settings.SkeletonESP then
                    local function Connect(L, P1, P2)
                        local Part1, Part2 = Char:FindFirstChild(P1), Char:FindFirstChild(P2)
                        if Part1 and Part2 then
                            local V1 = Camera:WorldToViewportPoint(Part1.Position)
                            local V2 = Camera:WorldToViewportPoint(Part2.Position)
                            L.Visible = true
                            L.From = Vector2.new(V1.X, V1.Y + _G.Settings.FixEsp)
                            L.To = Vector2.new(V2.X, V2.Y + _G.Settings.FixEsp)
                        else L.Visible = false end
                    end
                    Connect(Skeleton.H_T, "Head", "UpperTorso")
                    Connect(Skeleton.T_LA, "UpperTorso", "LeftUpperArm")
                    Connect(Skeleton.LA_LH, "LeftUpperArm", "LeftLowerArm")
                    Connect(Skeleton.T_RA, "UpperTorso", "RightUpperArm")
                    Connect(Skeleton.RA_RH, "RightUpperArm", "RightLowerArm")
                    Connect(Skeleton.T_LL, "LowerTorso", "LeftUpperLeg")
                    Connect(Skeleton.LL_LF, "LeftUpperLeg", "LeftLowerLeg")
                    Connect(Skeleton.T_RL, "LowerTorso", "RightUpperLeg")
                    Connect(Skeleton.RL_RF, "RightUpperLeg", "RightLowerLeg")
                else
                    for _, l in pairs(Skeleton) do l.Visible = false end
                end
            else
                Box.Visible = false; Line.Visible = false; Name.Visible = false
                for _, l in pairs(Skeleton) do l.Visible = false end
            end
        else
            Box.Visible = false; Line.Visible = false; Name.Visible = false
            for _, l in pairs(Skeleton) do l.Visible = false end
        end
    end)
end

for _, v in pairs(Players:GetPlayers()) do ApplyESP(v) end
Players.PlayerAdded:Connect(ApplyESP)
